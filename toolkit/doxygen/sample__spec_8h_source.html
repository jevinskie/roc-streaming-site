237 //! Parse sample spec from string.
238 //!
239 //! @remarks
240 //! The input string should have the form:
241 //! - "<format>/<rate>/<channels>"
242 //!
243 //! Where:
244 //! - "<format>" is string name of sample format (e.g. "s16")
245 //! - "<rate>" is a positive integer
246 //! - "<channels>" can be: "<surround preset>", "<surround channel list>",
247 //! "<multitrack mask>", "<multitrack channel list>"
248 //!
249 //! - "<surround preset>" is a string name of predefined surround channel
250 //! mask, e.g. "stereo", "surround4.1", etc.
251 //! - "<surround channel list>" is comma-separated list of surround channel names,
252 //! e.g. "FL,FC,FR"
253 //!
254 //! - "<multitrack mask>" is a 1024-bit hex mask defining which tracks are
255 //! enabled, e.g. "0xAA00BB00"
256 //! - "<multitrack channel list>" is a comma-separated list of track numbers
257 //! or ranges, e.g. "1,2,5-8"
258 //!
259 //! Each of the three components ("<format>", "<rate>", "<channels>") may be set
260 //! to "-", which means "keep unset".
261 //!
262 //! All four forms of "<channels>" component are alternative ways to represent a
263 //! bitmask of enabled channels or tracks. The order of channels does no matter.
264 //!
265 //! Examples:
266 //! - "s16/44100/stereo"
267 //! - "s18_4le/48000/FL,FC,FR"
268 //! - "f32/96000/1,2,10-20,31"
269 //! - "f32/96000/0xA0000000FFFF0000000C"
270 //! - "-/44100/-"
271 //! - "-/-/-"
272 //!
273 //! @returns
274 //! false if string can't be parsed.
275 ROC_ATTR_NODISCARD bool parse_sample_spec(const char* str, SampleSpec& result);
