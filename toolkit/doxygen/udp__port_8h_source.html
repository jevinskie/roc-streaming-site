32 //! UDP port parameters.
33 struct UdpConfig {
34  //! Port will bind to this address.
35  //! If IP is zero, INADDR_ANY is used, i.e. the socket is bound to all network
36  //! interfaces. If port is zero, a random free port is selected.
37  address::SocketAddr bind_address;
38 
39  //! If not empty, port will join multicast group on the interface
40  //! with given address. May be "0.0.0.0" or "[::]" to join on all interfaces.
41  //! Used only if receiving is started.
42  char multicast_interface[64];
43 
44  //! If set, enable SO_REUSEADDR when binding socket to non-ephemeral port.
45  //! If not set, SO_REUSEADDR is enabled only for multicast sockets when
46  //! binding to non-ephemeral port.
47  bool enable_reuseaddr;
48 
49  //! If true, allow non-blocking writes directly in write() method.
50  //! If non-blocking write can't be performed, port falls back to
51  //! regular asynchronous write.
52  //! Used only if sending is started.
53  bool enable_non_blocking;
54 
55  UdpConfig()
56  : enable_reuseaddr(false)
57  , enable_non_blocking(true) {
58  multicast_interface[0] = '\0';
59  }
60 
61  //! Check two configs for equality.
62  bool operator==(const UdpConfig& other) const {
63  return bind_address == other.bind_address
64  && strcmp(multicast_interface, other.multicast_interface) == 0
65  && enable_reuseaddr == other.enable_reuseaddr
66  && enable_non_blocking == other.enable_non_blocking;
67  }
68 };
