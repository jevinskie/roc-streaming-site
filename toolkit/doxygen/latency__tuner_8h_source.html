27 //! Latency tuner backend.
28 //! Defines which latency we monitor and tune to achieve target.
29 enum LatencyTunerBackend {
30  //! Deduce best default for given settings.
31  LatencyTunerBackend_Default,
32 
33  //! Latency is Network Incoming Queue length.
34  //! Calculated on receiver without use of any signaling protocol.
35  //! Reported back to sender via RTCP XR.
36  LatencyTunerBackend_Niq,
37 
38  //! Latency is End-to-end delay.
39  //! Can on receiver if RTCP XR is supported by both sides.
40  //! Reported back to sender via RTCP XR.
41  LatencyTunerBackend_E2e
42 };

44 //! Latency tuner profile.
45 //! Defines whether and how we tune latency on fly to compensate clock
46 //! drift and jitter.
47 enum LatencyTunerProfile {
48  //! Deduce best default for given settings.
49  LatencyTunerProfile_Default,
50 
51  //! Do not tune latency.
52  LatencyTunerProfile_Intact,
53 
54  //! Fast and responsive tuning.
55  //! Good for lower network latency and jitter.
56  LatencyTunerProfile_Responsive,
57 
58  //! Slow and smooth tuning.
59  //! Good for higher network latency and jitter.
60  LatencyTunerProfile_Gradual
61 };

63 //! Latency settings.
64 struct LatencyConfig {
65  //! Latency tuner backend to use.
66  //! @remarks
67  //! Defines which latency to monitor & tune.
68  LatencyTunerBackend tuner_backend;
69 
70  //! Latency tuner profile to use.
71  //! @remarks
72  //! Defines how smooth is the tuning.
73  LatencyTunerProfile tuner_profile;
74 
75  //! Target latency.
76  //! @remarks
77  //! Latency tuner will try to keep latency close to this value.
78  //! @note
79  //! If zero, default value is used if possible.
80  //! Negative value is an error.
81  core::nanoseconds_t target_latency;
82 
83  //! Minimum allowed latency.
84  //! @remarks
85  //! If the latency goes out of bounds, the session is terminated.
86  //! @note
87  //! If both min_latency and max_latency are zero, defaults are used.
88  core::nanoseconds_t min_latency;
89 
90  //! Maximum allowed latency.
91  //! @remarks
92  //! If the latency goes out of bounds, the session is terminated.
93  //! @note
94  //! If both min_latency and max_latency are zero, defaults are used.
95  core::nanoseconds_t max_latency;
96 
97  //! Maximum delay since last packet before queue is considered stalling.
98  //! @remarks
99  //! If niq_stalling becomes larger than stalling_tolerance, latency
100  //! tolerance checks are temporary disabled.
101  //! @note
102  //! If zero, default value is used if possible.
103  //! Negative value is an error.
104  core::nanoseconds_t stale_tolerance;
105 
106  //! Scaling update interval.
107  //! @remarks
108  //! How often to run FreqEstimator and update Resampler scaling.
109  //! @note
110  //! If zero, default value is used.
111  //! Negative value is an error.
112  core::nanoseconds_t scaling_interval;
113 
114  //! Maximum allowed deviation of freq_coeff from 1.0.
115  //! @remarks
116  //! If the scaling goes out of bounds, it is trimmed.
117  //! For example, 0.01 allows freq_coeff values in range [0.99; 1.01].
118  //! @note
119  //! If zero, default value is used.
120  //! Negative value is an error.
121  float scaling_tolerance;
122 
123  //! Initialize.
124  LatencyConfig()
125  : tuner_backend(LatencyTunerBackend_Default)
126  , tuner_profile(LatencyTunerProfile_Default)
127  , target_latency(0)
128  , min_latency(0)
129  , max_latency(0)
130  , stale_tolerance(0)
131  , scaling_interval(0)
132  , scaling_tolerance(0) {
133  }
134 
135  //! Automatically fill missing settings.
136  void deduce_defaults(core::nanoseconds_t default_target_latency, bool is_receiver);
137 };

139 //! Latency-related metrics.

161 //! Latency tuner.
162 //!
163 //! On receiver, LatencyMonitor computes local metrics and passes them to LatencyTuner.
164 //! On sender, FeedbackMonitor obtains remote metrics and passes them to LatencyTuner.
165 //! In both cases, LatencyTuner processes metrics and computes scaling factor that
166 //! should be passed to resampler.
167 //!
168 //! Features:
169 //! - monitors how close actual latency and target latency are
170 //! - monitors whether latency goes out of bounds
171 //! - assuming that the difference between actual latency and target latency is
172 //! caused by the clock drift between sender and receiver, calculates scaling
173 //! factor for resampler to compensate it
174 class LatencyTuner : public core::NonCopyable<> {
175 public:
176  //! Initialize.
177  LatencyTuner(const LatencyConfig& config, const SampleSpec& sample_spec);
178 
179  //! Check if the object was initialized successfully.
180  bool is_valid() const;
181 
182  //! Pass updated metrics to tuner.
183  //! Tuner will use new values next time when update_stream() is called.
184  void write_metrics(const LatencyMetrics& latency_metrics,
185  const packet::LinkMetrics& link_metrics);
186 
187  //! Update stream latency and scaling.
188  //! This method performs all actual work:
189  //! - depending on configured backend, selects which latency from
190  //! metrics to use
191  //! - check if latency goes out of bounds and session should be
192  //! terminated; if so, returns false
193  //! - computes updated scaling based on latency history and configured
194  //! profile
195  bool update_stream();
196 
197  //! Advance stream by given number of samples.
198  //! Should be called after updating stream.
199  void advance_stream(packet::stream_timestamp_t duration);
200 
201  //! Get computed scaling.
202  //! Latency tuner expects that this scaling will applied to the stream
203  //! resampler, so that the latency will slowly achieve target value.
204  //! Returned value is close to 1.0.
205  //! If no scaling was computed yet, it returns 0.0.
206  float get_scaling() const;
207 
208 private:
209  bool check_bounds_(packet::stream_timestamp_diff_t latency);
210  void compute_scaling_(packet::stream_timestamp_diff_t latency);
211  void report_();
212 
213  core::Optional<FreqEstimator> fe_;
214 
215  packet::stream_timestamp_t stream_pos_;
216 
217  packet::stream_timestamp_diff_t scale_interval_;
218  packet::stream_timestamp_t scale_pos_;

221  packet::stream_timestamp_t report_pos_;

223  float freq_coeff_;
224  const float freq_coeff_max_delta_;
225 
226  const LatencyTunerBackend backend_;
227  const LatencyTunerProfile profile_;
228 
229  const bool enable_tuning_;
230  const bool enable_bounds_;
231 
232  bool has_niq_latency_;
233  packet::stream_timestamp_diff_t niq_latency_;

244  packet::stream_timestamp_diff_t max_latency_;
245  packet::stream_timestamp_diff_t max_stalling_;
246 
247  const SampleSpec sample_spec_;
248 
249  bool valid_;
