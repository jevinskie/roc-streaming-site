40 //! Defines multiplication matrix for two channel masks.
41 //! Instead of defining the whole matrix, it defines a list of pairs of
42 //! output and input channel numbers and corresponding coefficients.
43 //! Such representation allows more compact definition in the source
44 //! code. The actual matrix is built by channel mapper at runtime.
45 struct ChannelMapTable {
46  const char* name; //!< Mapping name.
47 
48  ChannelMask in_mask; //!< Channel mask of input stream.
49  ChannelMask out_mask; //!< Channel mask of output stream.
50 
51  //! Transformation rules.
52  //! Rules are used to fill channel mapping matrix.
53  ChannelMapRule rules[24];
54 };
55 
56 //! Defines ordered list of channels.
